class Solution:
    
    def rotate(self, nums: List[int], k: int) -> None:
        n=len(nums)
        k=k%n
        temp=[]
        for i in range(n-k):
            temp.append(nums[i])
        for i in range(n-k,n):
            nums[i-(n-k)]=nums[i]
        for i in range(k,n):
            nums[i]=temp[i-k]
        return nums
